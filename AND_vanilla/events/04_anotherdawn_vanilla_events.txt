#Spontaneous religious spawning
province_event = {
	id = AND.1
	desc = AND.1.desc
	picture = GFX_evt_council
	
	trigger = {
		religion = ancestry
		NOT = {
			any_neighbor_province = {
				NOT = {
					religion = placeholder1
				}
			}
		}
		NOT = {
			any_independent_ruler = {
				any_realm_province = {
					religion = placeholder1
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = AND.1.a
		ROOT = {
			religion = placeholder1
		}
		ROOT = {
			owner = {
				character_event = { id = AND.16 }
			}
		}
	}
}
